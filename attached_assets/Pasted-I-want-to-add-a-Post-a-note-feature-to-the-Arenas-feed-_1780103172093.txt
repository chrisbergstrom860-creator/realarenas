javascriptapp.post('/api/posts/:id/like', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data: existing } = await supabaseAdmin
    .from('post_likes')
    .select()
    .eq('post_id', req.params.id)
    .eq('user_id', req.user.id)
    .single();
  if (existing) {
    await supabaseAdmin.from('post_likes').delete()
      .eq('post_id', req.params.id)
      .eq('user_id', req.user.id);
    return res.json({ liked: false });
  }
  await supabaseAdmin.from('post_likes').insert({
    post_id: req.params.id,
    user_id: req.user.id
  });
  res.json({ liked: true });
});

7. Add these JavaScript functions to the existing script block in arenas-feed.html:
javascript// ── POST COMPOSER ──
let composerSport = null;
let composerFeeling = null;

function updateComposer(val) {
  const bottom = document.getElementById('composer-bottom');
  const btn = document.getElementById('composer-post-btn');
  const counter = document.getElementById('char-count');
  const remaining = 280 - val.length;
  
  bottom.style.display = val.length > 0 ? 'block' : 'none';
  btn.disabled = val.trim().length === 0;
  counter.textContent = remaining;
  counter.className = 'char-count' + (remaining < 50 ? ' warning' : '') + (remaining < 20 ? ' danger' : '');

  // Auto-resize textarea
  const ta = document.getElementById('composer-input');
  ta.style.height = 'auto';
  ta.style.height = Math.min(ta.scrollHeight, 120) + 'px';
}

function selectComposerSport(btn, sport) {
  document.querySelectorAll('.c-sport-chip').forEach(c => c.classList.remove('selected'));
  if (composerSport === sport) {
    composerSport = null;
  } else {
    btn.classList.add('selected');
    composerSport = sport;
  }
}

function selectFeeling(btn, feeling) {
  document.querySelectorAll('.c-feel-chip').forEach(c => c.classList.remove('selected'));
  if (composerFeeling === feeling) {
    composerFeeling = null;
  } else {
    btn.classList.add('selected');
    composerFeeling = feeling;
  }
}

function cancelPost() {
  document.getElementById('composer-input').value = '';
  document.getElementById('composer-bottom').style.display = 'none';
  document.getElementById('char-count').textContent = '280';
  document.querySelectorAll('.c-sport-chip, .c-feel-chip').forEach(c => c.classList.remove('selected'));
  composerSport = null;
  composerFeeling = null;
}

async function submitPost() {
  const content = document.getElementById('composer-input').value.trim();
  if (!content) return;
  
  const btn = document.getElementById('composer-post-btn');
  btn.textContent = 'Posting…';
  btn.disabled = true;

  try {
    const response = await fetch('/api/posts/create', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        content,
        sport: composerSport,
        feeling: composerFeeling
      })
    });
    const data = await response.json();
    
    if (data.error) {
      showToast('Error: ' + data.error);
      btn.textContent = 'Post note';
      btn.disabled = false;
      return;
    }

    // Add the new post to the top of the feed
    addPostToFeed(data.post);
    cancelPost();
    showToast('✓ Note posted — your followers can see it now!');
  } catch (err) {
    showToast('Something went wrong — please try again');
    btn.textContent = 'Post note';
    btn.disabled = false;
  }
}

async function likePost(btn, postId) {
  try {
    const response = await fetch(`/api/posts/${postId}/like`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' }
    });
    const data = await response.json();
    const currentLikes = parseInt(btn.textContent.match(/\d+/)?.[0] || '0');
    if (data.liked) {
      btn.classList.add('liked');
      btn.textContent = '👍 ' + (currentLikes + 1) + ' kudos';
    } else {
      btn.classList.remove('liked');
      btn.textContent = '👍 ' + Math.max(0, currentLikes - 1) + ' kudos';
    }
  } catch (err) {
    showToast('Could not like post — please try again');
  }
}

function getTimeAgo(date) {
  const seconds = Math.floor((new Date() - date) / 1000);
  if (seconds < 60) return 'Just now';
  if (seconds < 3600) return Math.floor(seconds/60) + 'm ago';
  if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
  return Math.floor(seconds/86400) + 'd ago';
}
