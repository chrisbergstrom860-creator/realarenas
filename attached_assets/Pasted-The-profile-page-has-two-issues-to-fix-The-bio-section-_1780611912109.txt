Fourth — add a /api/profile/update POST route to server.js so the save button actually writes to Supabase:
javascriptapp.post('/api/profile/update', requireAuth, async (req, res) => {
  const { name, handle, location, bio, sports, level } = req.body;
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { error } = await supabaseAdmin
    .from('profiles')
    .upsert({
      id: req.user.id,
      name,
      handle,
      location,
      bio,
      sports,
      level
    });
  if (error) return res.json({ error: error.message });
  res.json({ success: true });
});
