javascript// Get events for the logged in user
app.get('/api/events', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const userId = req.user.id;

  // Get user's club memberships
  const { data: memberships } = await supabaseAdmin
    .from('memberships')
    .select('club_id')
    .eq('user_id', userId);
  const clubIds = (memberships || []).map(m => m.club_id);

  // Get who the user follows
  const { data: following } = await supabaseAdmin
    .from('follows')
    .select('following_id')
    .eq('follower_id', userId);
  const followingIds = (following || []).map(f => f.following_id);

  // Get upcoming public events
  const { data: upcomingEvents } = await supabaseAdmin
    .from('events')
    .select('*, profiles:created_by(name, handle)')
    .eq('visibility', 'public')
    .gte('date', new Date().toISOString())
    .order('date', { ascending: true })
    .limit(50);

  // Get club events
  const { data: clubEvents } = await supabaseAdmin
    .from('events')
    .select('*, profiles:created_by(name, handle), clubs:club_id(name)')
    .in('club_id', clubIds.length ? clubIds : ['00000000-0000-0000-0000-000000000000'])
    .gte('date', new Date().toISOString())
    .order('date', { ascending: true });

  // Get events created by the user
  const { data: myCreatedEvents } = await supabaseAdmin
    .from('events')
    .select('*, profiles:created_by(name, handle)')
    .eq('created_by', userId)
    .order('date', { ascending: true });

  // Get all event IDs to fetch RSVPs
  const allEventIds = [
    ...(upcomingEvents || []).map(e => e.id),
    ...(clubEvents || []).map(e => e.id),
    ...(myCreatedEvents || []).map(e => e.id)
  ].filter((v, i, a) => a.indexOf(v) === i);

  // Get all RSVPs for these events
  const { data: allRsvps } = await supabaseAdmin
    .from('event_rsvps')
    .select('event_id, user_id, status, profiles:user_id(name, handle)')
    .in('event_id', allEventIds.length ? allEventIds : ['00000000-0000-0000-0000-000000000000']);

  // Get user's own RSVPs
  const { data: myRsvps } = await supabaseAdmin
    .from('event_rsvps')
    .select('event_id, status, events(id, title, date, location, sport)')
    .eq('user_id', userId)
    .order('created_at', { ascending: false });

  // Enrich events with RSVP data
  function enrichEvent(event) {
    const eventRsvps = (allRsvps || []).filter(r => r.event_id === event.id);
    const goingCount = eventRsvps.filter(r => r.status === 'going').length;
    const interestedCount = eventRsvps.filter(r => r.status === 'interested').length;
    const myRsvp = eventRsvps.find(r => r.user_id === userId);

    // Find followers who are going
    const followersGoing = eventRsvps.filter(r =>
      r.status === 'going' && followingIds.includes(r.user_id)
    );

    return {
      ...event,
      goingCount,
      interestedCount,
      myRsvpStatus: myRsvp?.status || null,
      followersGoing: followersGoing.map(r => ({
        name: r.profiles?.name || 'Athlete',
        handle: r.profiles?.handle || 'athlete'
      })),
      isOwner: event.created_by === userId
    };
  }

  res.json({
    upcomingEvents: (upcomingEvents || []).map(enrichEvent),
    clubEvents: (clubEvents || []).map(enrichEvent),
    myCreatedEvents: (myCreatedEvents || []).map(enrichEvent),
    myRsvps: myRsvps || []
  });
});

app.post('/api/events/create', requireAuth, async (req, res) => {

  }
  const { data: event, error } = await supabaseAdmin
    .from('events')
    .insert({
      created_by: req.user.id,
      club_id: club_id || null,
      title, sport, event_type,
      date: new Date(date).toISOString(),
      location, distance,
      max_participants: max_participants ? parseInt(max_participants) : null,
      entry_fee, level, description,
      visibility: visibility || 'public'
    })
    .select()
    .single();
  if (error) return res.json({ error: error.message });
  // Auto RSVP the creator as going
  await supabaseAdmin.from('event_rsvps').insert({
    event_id: event.id,
    user_id: req.user.id,
    status: 'going'
  });
  // Get creator profile
  const { data: profile } = await supabaseAdmin
    .from('profiles')
    .select('name')
    .eq('id', req.user.id)
    .single();
  // Send invites to selected followers
  for (const inviteeId of (invitees || [])) {

      userId: inviteeId,
      type: 'event',
      title: 'Event invite',

      link: '/events',
      actorId: req.user.id,
      entityId: event.id

  }
  // Notify club members if posted to a club
  if (club_id) {

        actorId: req.user.id,
        entityId: event.id

app.post('/api/events/:id/rsvp', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { status } = req.body;
  if (!['going', 'interested', 'cancelled'].includes(status)) {
    return res.json({ error: 'Invalid status' });
  }
  // Check if already RSVPd
  const { data: existing } = await supabaseAdmin
    .from('event_rsvps')
    .select()
    .eq('event_id', req.params.id)
    .eq('user_id', req.user.id)
    .single();
  if (existing) {
    if (status === 'cancelled') {
      await supabaseAdmin.from('event_rsvps').delete()
        .eq('event_id', req.params.id)
        .eq('user_id', req.user.id);
    } else {
      await supabaseAdmin.from('event_rsvps').update({ status })
        .eq('event_id', req.params.id)
        .eq('user_id', req.user.id);
    }
  } else {
    await supabaseAdmin.from('event_rsvps').insert({
      event_id: req.params.id,
      user_id: req.user.id,
      status
    });
  }
  // Notify event creator when someone RSVPs as going
  if (status === 'going') {
    const { data: event } = await supabaseAdmin
      .from('events')
      .select('created_by, title')
      .eq('id', req.params.id)
      .single();
    const { data: profile } = await supabaseAdmin
      .from('profiles')
      .select('name')
      .eq('id', req.user.id)
      .single();
    if (event && event.created_by !== req.user.id) {
      await createNotification(supabaseAdmin, {
        userId: event.created_by,
        type: 'event',
        title: 'New RSVP',
        body: `${profile?.name || 'Someone'} is going to your event "${event.title}"`,
        link: '/events',
        actorId: req.user.id,
        entityId: req.params.id
      });
    }
    // Notify followers that this user is going
    const { data: followers } = await supabaseAdmin
      .from('follows')
      .select('follower_id')
      .eq('following_id', req.user.id);
    for (const follower of (followers || [])) {
      await createNotification(supabaseAdmin, {
        userId: follower.follower_id,
        type: 'event',
        title: 'Friend going to an event',
        body: `${profile?.name || 'Someone you follow'} is going to "${event?.title}" — are you in?`,
        link: '/events',
        actorId: req.user.id,
        entityId: req.params.id
      });
    }
  }
  res.json({ success: true, status });
});

  function groupByTime(events) {
    const groups = {};
    events.forEach(e => {

      if (!groups[label]) groups[label] = [];
      groups[label].push(e);
    });
    return groups;
  }
