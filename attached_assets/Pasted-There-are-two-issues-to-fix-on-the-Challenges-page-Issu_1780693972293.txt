Please fix the /api/challenges route in server.js so that:

The My Challenges query fetches challenges where created_by = userId OR where the user has a row in challenge_participants with their user_id. This should be two separate queries merged together — one for created challenges and one for joined challenges — to avoid the Supabase .or() syntax issues with UUIDs:

javascript// Get challenges the user created
const { data: createdChallenges } = await supabaseAdmin
  .from('challenges')
  .select('*, profiles:created_by(name, handle)')
  .eq('created_by', userId)
  .order('created_at', { ascending: false });

// Get challenges the user joined (but didn't create)
const joinedIds = myIds.filter(id =>
  !(createdChallenges || []).find(c => c.id === id)
);
let joinedChallenges = [];
if (joinedIds.length > 0) {
  const { data } = await supabaseAdmin
    .from('challenges')
    .select('*, profiles:created_by(name, handle)')
    .in('id', joinedIds)
    .order('created_at', { ascending: false });
  joinedChallenges = data || [];
}

// Combine into myChallenges
const myChallenges = [...(createdChallenges || []), ...joinedChallenges];

The Discover/public challenges query should EXCLUDE challenges the user created or already joined:

javascriptconst excludeFromDiscover = [
  ...(myChallenges || []).map(c => c.id),
  ...myIds
].filter((v, i, a) => a.indexOf(v) === i);

const { data: publicChallenges } = await supabaseAdmin
  .from('challenges')
  .select('*, profiles:created_by(name, handle)')
  .eq('visibility', 'public')
  .gt('end_date', new Date().toISOString())
  .not('id', 'in', `(${excludeFromDiscover.join(',')})`)
  .order('created_at', { ascending: false })

If excludeFromDiscover is empty skip the .not() filter entirely to avoid a Supabase error.
