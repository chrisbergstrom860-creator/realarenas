javascript// Profile overview — this week summary + day strip
app.get('/api/profile/overview', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const userId = req.user.id;
  const now = new Date();
  const day = now.getDay() || 7;
  const weekStart = new Date(now);
  weekStart.setDate(now.getDate() - day + 1);
  weekStart.setHours(0, 0, 0, 0);

  // This week's activities
  const { data: weekActs } = await supabaseAdmin
    .from('activities')
    .select('sport, distance, duration, date')
    .eq('user_id', userId)
    .gte('date', weekStart.toISOString());
  const acts = weekActs || [];
  function km(a) {
    const d = parseFloat((a.distance || '0').replace(/[^0-9.]/g, ''));
    return isNaN(d) ? 0 : d;
  }
  const weekKm = Math.round(acts.reduce((s, a) => s + km(a), 0) * 10) / 10;
  const weekHours = Math.round(acts.reduce((s, a) => s + parseDurationHours(a.duration), 0) * 10) / 10;
  const weekPoints = calculatePoints(acts);

  // Day strip — which days this week had activity
  const dayLabels = ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'];
  const activeDaySet = new Set(acts.map(a => {
    const d = new Date(a.date);
    return (d.getDay() || 7) - 1; // 0=Mon
  }));
  const todayIdx = day - 1;
  const dayStrip = dayLabels.map((label, i) => ({
    label,
    state: i === todayIdx && !activeDaySet.has(i) ? 'today'
      : activeDaySet.has(i) ? 'active'
      : i < todayIdx ? 'rest' : 'future'
  }));

  // Current streak (all-time activities for streak calc)
  const { data: allActs } = await supabaseAdmin
    .from('activities')
    .select('date')
    .eq('user_id', userId);
  const days = [...new Set((allActs || []).map(a => new Date(a.date).toDateString()))]
    .map(d => new Date(d)).sort((a, b) => a - b);
  let currentStreak = 0;
  if (days.length > 0) {
    const today = new Date(); today.setHours(0,0,0,0);
    const last = new Date(days[days.length - 1]); last.setHours(0,0,0,0);
    if (Math.round((today - last) / 86400000) <= 1) {
      currentStreak = 1;
      for (let i = days.length - 1; i > 0; i--) {
        if (Math.round((days[i] - days[i-1]) / 86400000) === 1) currentStreak++;
        else break;
      }
    }
  }

  // Recent activities (last 3)
  const { data: recentActs } = await supabaseAdmin
    .from('activities')
    .select('id, sport, title, distance, duration, pace, created_at')
    .eq('user_id', userId)
    .order('created_at', { ascending: false })
    .limit(3);

  // My active challenges with progress + rank
  const { data: myParticipations } = await supabaseAdmin
    .from('challenge_participants')
    .select('challenge_id, challenges(id, title, sport, goal_type, goal_target, goal_unit, start_date, end_date)')
    .eq('user_id', userId);
  const activeChallenges = [];
  for (const cp of (myParticipations || [])) {
    const ch = cp.challenges;
    if (!ch || new Date(ch.end_date) < now) continue;
    // My progress
    const { data: chActs } = await supabaseAdmin
      .from('activities')
      .select('sport, distance, date')
      .eq('user_id', userId)
      .gte('date', ch.start_date)
      .lte('date', ch.end_date);
    let progress = 0;
    const uniqueDays = new Set();
    (chActs || []).forEach(a => {
      if (ch.sport !== 'any' && a.sport !== ch.sport) return;
      if (ch.goal_type === 'distance') progress += km(a);
      else if (ch.goal_type === 'streak') uniqueDays.add(new Date(a.date).toDateString());
      else progress += 1;
    });
    if (ch.goal_type === 'streak') progress = uniqueDays.size;
    progress = Math.round(progress * 10) / 10;
    // My rank — count participants with more progress (lightweight: total participants only)
    const { count: totalParticipants } = await supabaseAdmin
      .from('challenge_participants')
      .select('*', { count: 'exact', head: true })
      .eq('challenge_id', ch.id);
    const daysLeft = Math.max(0, Math.ceil((new Date(ch.end_date) - now) / 86400000));
    const pct = ch.goal_target > 0 ? Math.min(100, Math.round((progress / ch.goal_target) * 100)) : 0;
    // Pace check: expected pct based on elapsed time
    const totalDays = Math.max(1, (new Date(ch.end_date) - new Date(ch.start_date)) / 86400000);
    const elapsedDays = Math.max(0, (now - new Date(ch.start_date)) / 86400000);
    const expectedPct = Math.round((elapsedDays / totalDays) * 100);
    let statusText, statusColor;
    if (pct >= 100) { statusText = 'Goal achieved ✓'; statusColor = '#10B981'; }
    else if (ch.goal_type === 'streak') {
      const remaining = ch.goal_target - progress;
      statusText = remaining <= 2 ? `${remaining} more day${remaining !== 1 ? 's' : ''} — don't break it!` : `${remaining} days to go`;
      statusColor = '#854D0E';
    }
    else if (pct >= expectedPct) { statusText = 'On pace'; statusColor = '#10B981'; }
    else { statusText = 'Behind pace — push on'; statusColor = '#854D0E'; }
    activeChallenges.push({
      id: ch.id, title: ch.title, sport: ch.sport,
      progress, target: ch.goal_target, unit: ch.goal_unit,
      pct, daysLeft, statusText, statusColor,
      totalParticipants: totalParticipants || 0
    });
  }

  // My upcoming RSVPs
  const { data: myRsvps } = await supabaseAdmin
    .from('event_rsvps')
    .select('status, events(id, title, date, location)')
    .eq('user_id', userId)
    .in('status', ['going', 'interested']);
  const upcomingRsvps = (myRsvps || [])
    .filter(r => r.events && new Date(r.events.date) >= now)
    .sort((a, b) => new Date(a.events.date) - new Date(b.events.date))
    .slice(0, 3)
    .map(r => ({ status: r.status, ...r.events }));
  // Going counts for those events
  for (const ev of upcomingRsvps) {
    const { count } = await supabaseAdmin
      .from('event_rsvps')
      .select('*', { count: 'exact', head: true })
      .eq('event_id', ev.id)
      .eq('status', 'going');
    ev.goingCount = count || 0;
  }

  res.json({
    week: { activities: acts.length, km: weekKm, hours: weekHours, points: weekPoints },
    dayStrip,
    currentStreak,
    recentActivities: recentActs || [],
    activeChallenges,
    upcomingRsvps
  });
});
