// Get activities for a user
app.get('/api/activities/:userId', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data, error } = await supabaseAdmin
    .from('activities')
    .select('*')
    .eq('user_id', req.params.userId)
    .order('date', { ascending: false })
    .limit(50);
  if (error) return res.json({ error: error.message });
  res.json({ activities: data || [] });
});

// Delete an activity
app.delete('/api/activities/:id', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { error } = await supabaseAdmin
    .from('activities')
    .delete()
    .eq('id', req.params.id)
    .eq('user_id', req.user.id);
  if (error) return res.json({ error: error.message });
  res.json({ success: true });
});

// Get activities for the feed (from followed users)
app.get('/api/feed/activities', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data: following } = await supabaseAdmin
    .from('follows')
    .select('following_id')
    .eq('follower_id', req.user.id);
  const followingIds = [...(following || []).map(f => f.following_id), req.user.id];
  const { data, error } = await supabaseAdmin
    .from('activities')
    .select('*, profiles:user_id(name, handle)')
    .in('user_id', followingIds)
    .order('date', { ascending: false })
    .limit(20);
  if (error) return res.json({ error: error.message });
  res.json({ activities: data || [] });
});

function toggleActivityForm() {
  const form = document.getElementById('activity-form');
  const isVisible = form.style.display !== 'none';
  form.style.display = isVisible ? 'none' : 'block';
  if (!isVisible) {
    const today = new Date().toISOString().split('T')[0];
    document.getElementById('act-date').value = today;
    document.getElementById('act-title').value = '';
    document.getElementById('act-duration').value = '';
    document.getElementById('act-notes').value = '';
    document.querySelectorAll('.act-sport-chip, .act-feeling-chip').forEach(c => c.classList.remove('selected'));
    document.getElementById('act-sport-fields').style.display = 'none';
    document.getElementById('act-error').style.display = 'none';
    selectedActivitySport = null;
    selectedActivityFeeling = null;
    form.scrollIntoView({ behavior: 'smooth', block: 'start' });
  }
}

function selectActivitySport(btn, sport) {
  document.querySelectorAll('.act-sport-chip').forEach(c => c.classList.remove('selected'));
  btn.classList.add('selected');
  selectedActivitySport = sport;
  const config = activitySportFields[sport];
  if (config) {
    document.getElementById('act-sport-fields-label').textContent = config.label;
    document.getElementById('act-sport-fields-body').innerHTML = config.html;
    document.getElementById('act-sport-fields').style.display = 'block';
  }
}

function selectFeeling(btn, feeling) {
  document.querySelectorAll('.act-feeling-chip').forEach(c => c.classList.remove('selected'));
  if (selectedActivityFeeling === feeling) {
    selectedActivityFeeling = null;
  } else {
    btn.classList.add('selected');
    selectedActivityFeeling = feeling;
  }
}

function showActError(msg) {
  const el = document.getElementById('act-error');
  el.textContent = msg;
  el.style.display = 'block';
  el.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
  setTimeout(() => el.style.display = 'none', 4000);
}

async function saveActivity() {
  const sport = selectedActivitySport;
  const title = document.getElementById('act-title')?.value?.trim();
  const date = document.getElementById('act-date')?.value;
  const duration = document.getElementById('act-duration')?.value?.trim();
  const notes = document.getElementById('act-notes')?.value?.trim();

  if (!sport) { showActError('Please select a sport.'); return; }
  if (!title) { showActError('Please enter an activity title.'); return; }
  if (!duration) { showActError('Please enter the duration.'); return; }

  const btn = document.getElementById('save-activity-btn');
  btn.textContent = 'Saving…';
  btn.disabled = true;

  const body = {
    sport, title, duration, notes,
    date: date ? new Date(date).toISOString() : new Date().toISOString(),
    feeling: selectedActivityFeeling,
    // Core optional fields
    distance: document.getElementById('sf-distance')?.value || null,
    pace: document.getElementById('sf-pace')?.value || null,
    avg_hr: document.getElementById('sf-hr')?.value || null,
    elevation: document.getElementById('sf-elev')?.value || null,
    cadence: document.getElementById('sf-cad')?.value || null,
    run_type: document.getElementById('sf-runtype')?.value || null,
    avg_power: document.getElementById('sf-power')?.value || null,
    avg_speed: document.getElementById('sf-speed')?.value || null,
    ride_type: document.getElementById('sf-ridetype')?.value || null,
    top_grade: document.getElementById('sf-grade')?.value || null,
    project_grade: document.getElementById('sf-project')?.value || null,
    problems_count: document.getElementById('sf-problems')?.value || null,
    climbing_style: document.getElementById('sf-style')?.value || null,
    climb_location: document.getElementById('sf-location')?.value || null,
    swim_pace: document.getElementById('sf-pace')?.value || null,
    pool_type: document.getElementById('sf-pooltype')?.value || null,
    stroke: document.getElementById('sf-stroke')?.value || null,
    session_type: document.getElementById('sf-sessiontype')?.value || null,
    position: document.getElementById('sf-position')?.value || null,
    session_focus: document.getElementById('sf-focus')?.value || null,
    total_volume: document.getElementById('sf-volume')?.value || null,
    top_lift: document.getElementById('sf-toplift')?.value || null,
    sets_completed: document.getElementById('sf-sets')?.value || null,
    rpe: document.getElementById('sf-rpe')?.value || null,
    trail: document.getElementById('sf-trail')?.value || null,
    terrain: document.getElementById('sf-terrain')?.value || null,
    pack_weight: document.getElementById('sf-pack')?.value || null,
    yoga_style: document.getElementById('sf-style')?.value || null,
    yoga_format: document.getElementById('sf-format')?.value || null,
    focus_area: document.getElementById('sf-focus')?.value || null,
    instructor: document.getElementById('sf-instructor')?.value || null
  };

  try {
    const response = await fetch('/api/activities/create', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body)
    });
    const result = await response.json();
    if (result.error) {
      showActError('Error saving: ' + result.error);
      btn.textContent = 'Save activity →';
      btn.disabled = false;
      return;
    }
    showToast('✓ Activity logged! Your followers can see it now.');
    toggleActivityForm();
    loadActivities();
  } catch (err) {
    showActError('Something went wrong — please try again.');
    btn.textContent = 'Save activity →';
    btn.disabled = false;
  }
}

async function loadActivities() {
  if (!window.ARENAS_DATA?.userId) return;
  try {
    const response = await fetch('/api/activities/' + window.ARENAS_DATA.userId);
    const data = await response.json();
    if (data.error || !data.activities) return;
    renderActivities(data.activities);
  } catch (err) {
    console.log('Could not load activities:', err);
  }
}

async function deleteActivity(id, card) {
  if (!confirm('Delete this activity? This cannot be undone.')) return;
  try {
    const response = await fetch('/api/activities/' + id, { method: 'DELETE' });
    const result = await response.json();
    if (result.success) {
      card.style.transition = 'opacity .25s, max-height .3s .1s';
      card.style.opacity = '0';
      setTimeout(() => card.remove(), 350);
      showToast('Activity deleted');
    }
  } catch (err) {
    showToast('Could not delete — please try again');
  }
}

  function timeAgo(dateStr) {
    const seconds = Math.floor((new Date() - new Date(dateStr)) / 1000);
    if (seconds < 60) return 'Just now';
    if (seconds < 3600) return Math.floor(seconds/60) + 'm ago';
    if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
    if (seconds < 604800) return Math.floor(seconds/86400) + 'd ago';
    return new Date(dateStr).toLocaleDateString();
  }

  function buildStats(activity) {

          <div class="pn-footer">
            <button class="pn-action">👍 Kudos</button>

            <span style="margin-left:auto;font-size:11px;color:#9CA3AF">Activity</span>
          </div>
        </div>
      </div>`;
  }).join('');
  if (feedContainer) feedContainer.insertAdjacentHTML('afterbegin', activityCardsHTML);
}
