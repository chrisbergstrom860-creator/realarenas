javascript// Get all challenges for the logged in user
// — their own, ones they joined, club challenges, and public ones
app.get('/api/challenges', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const userId = req.user.id;

  // Get challenges the user created or joined
  const { data: myParticipations } = await supabaseAdmin
    .from('challenge_participants')
    .select('challenge_id')
    .eq('user_id', userId);
  const myIds = (myParticipations || []).map(p => p.challenge_id);

  // Get user's club memberships
  const { data: memberships } = await supabaseAdmin
    .from('memberships')
    .select('club_id')
    .eq('user_id', userId);
  const clubIds = (memberships || []).map(m => m.club_id);

  // Get my challenges — ones I created or joined
  const { data: myChallenges } = await supabaseAdmin
    .from('challenges')
    .select('*, profiles:created_by(name, handle)')
    .or(`created_by.eq.${userId}${myIds.length ? ',id.in.(' + myIds.join(',') + ')' : ''}`)
    .order('created_at', { ascending: false });

  // Get club challenges from clubs the user belongs to
  const { data: clubChallenges } = await supabaseAdmin
    .from('challenges')
    .select('*, profiles:created_by(name, handle), clubs:club_id(name)')
    .in('club_id', clubIds.length ? clubIds : ['00000000-0000-0000-0000-000000000000'])
    .order('created_at', { ascending: false });

  // Get public challenges for discover tab
  const { data: publicChallenges } = await supabaseAdmin
    .from('challenges')
    .select('*, profiles:created_by(name, handle)')
    .eq('visibility', 'public')
    .gt('end_date', new Date().toISOString())
    .order('created_at', { ascending: false })
    .limit(20);

  // Get participant counts for all challenges
  const allIds = [
    ...(myChallenges || []).map(c => c.id),
    ...(clubChallenges || []).map(c => c.id),
    ...(publicChallenges || []).map(c => c.id)
  ].filter((v, i, a) => a.indexOf(v) === i);

  const { data: allParticipants } = await supabaseAdmin
    .from('challenge_participants')
    .select('challenge_id, user_id')
    .in('challenge_id', allIds.length ? allIds : ['00000000-0000-0000-0000-000000000000']);

  // Calculate progress for each challenge the user joined
  const progressMap = {};
  for (const challengeId of myIds) {
    const challenge = [...(myChallenges || []), ...(clubChallenges || [])]
      .find(c => c.id === challengeId);
    if (!challenge) continue;
    const { data: activities } = await supabaseAdmin
      .from('activities')
      .select('distance, duration, sport, date')
      .eq('user_id', userId)
      .gte('date', challenge.start_date)
      .lte('date', challenge.end_date);
    let progress = 0;
    if (challenge.goal_type === 'distance') {
      (activities || []).forEach(a => {
        if (challenge.sport === 'any' || a.sport === challenge.sport) {
          const dist = parseFloat((a.distance || '0').replace(/[^0-9.]/g, ''));
          if (!isNaN(dist)) progress += dist;
        }
      });
    } else if (challenge.goal_type === 'sessions') {
      progress = (activities || []).filter(a =>
        challenge.sport === 'any' || a.sport === challenge.sport
      ).length;
    } else if (challenge.goal_type === 'streak') {
      const dates = [...new Set((activities || [])
        .filter(a => challenge.sport === 'any' || a.sport === challenge.sport)
        .map(a => new Date(a.date).toDateString()))];
      progress = dates.length;
    }
    progressMap[challengeId] = Math.round(progress * 10) / 10;
  }

  // Enrich challenges with participant counts and join status
  function enrich(challenges) {
    return (challenges || []).map(c => {
      const participants = (allParticipants || []).filter(p => p.challenge_id === c.id);
      const isJoined = myIds.includes(c.id) || c.created_by === userId;
      const daysLeft = Math.max(0, Math.ceil(
        (new Date(c.end_date) - new Date()) / (1000 * 60 * 60 * 24)
      ));
      return {
        ...c,
        participantCount: participants.length,
        isJoined,
        progress: progressMap[c.id] || 0,
        daysLeft,
        isExpired: new Date(c.end_date) < new Date(),
        isOwner: c.created_by === userId
      };
    });
  }

  res.json({
    myChallenges: enrich(myChallenges),
    clubChallenges: enrich(clubChallenges),
    publicChallenges: enrich(publicChallenges),
    myJoinedIds: myIds
  });
});

// Create a challenge
app.post('/api/challenges/create', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const {
    title, description, sport, goal_type,
    goal_target, goal_unit, start_date,
    end_date, visibility, invitees, club_id
  } = req.body;
  if (!title || !goal_type || !goal_target || !start_date || !end_date) {
    return res.json({ error: 'Missing required fields' });
  }
  const { data: challenge, error } = await supabaseAdmin
    .from('challenges')
    .insert({
      created_by: req.user.id,
      club_id: club_id || null,
      title, description,
      sport: sport || 'any',
      goal_type, goal_target, goal_unit,
      start_date, end_date,
      visibility: visibility || 'public'
    })
    .select()
    .single();
  if (error) return res.json({ error: error.message });
  // Auto-join the creator
  await supabaseAdmin.from('challenge_participants').insert({
    challenge_id: challenge.id,
    user_id: req.user.id
  });
  // Send invites to followers
  const { data: profile } = await supabaseAdmin
    .from('profiles')
    .select('name')
    .eq('id', req.user.id)
    .single();
  for (const inviteeId of (invitees || [])) {
    await createNotification(supabaseAdmin, {
      userId: inviteeId,
      type: 'challenge',
      title: 'Challenge invite',
      body: `${profile?.name || 'Someone'} invited you to join "${title}" — ${goal_target} ${goal_unit} ${goal_type} challenge`,
      link: '/challenges',
      actorId: req.user.id,
      entityId: challenge.id
    });
  }
  res.json({ success: true, challenge });
});

  window.submitCreateChallenge = async function() {
    const title = document.getElementById('ch-title')?.value?.trim();
    const desc = document.getElementById('ch-desc')?.value?.trim();
    const sport = document.querySelector('.ch-sport-chip.selected')?.dataset?.sport || 'any';
    const goalType = document.querySelector('.ch-goal-type.selected')?.dataset?.type || 'distance';
    const goalTarget = document.getElementById('ch-target')?.value;
    const goalUnit = document.getElementById('ch-unit')?.value;
    const startDate = document.getElementById('ch-start')?.value;
    const endDate = document.getElementById('ch-end')?.value;
    const visibility = document.querySelector('.ch-vis-opt.selected')?.dataset?.vis || 'public';
    const inviteSelect = document.getElementById('ch-invite-select');
    const invitees = inviteSelect ? [...inviteSelect.selectedOptions].map(o => o.value) : [];
    const errorEl = document.getElementById('ch-error');

    if (!title) { errorEl.textContent = 'Please enter a challenge title.'; errorEl.style.display='block'; return; }
    if (!goalTarget || parseInt(goalTarget) < 1) { errorEl.textContent = 'Please enter a valid target.'; errorEl.style.display='block'; return; }
    if (!startDate || !endDate) { errorEl.textContent = 'Please set a start and end date.'; errorEl.style.display='block'; return; }
    if (new Date(endDate) <= new Date(startDate)) { errorEl.textContent = 'End date must be after start date.'; errorEl.style.display='block'; return; }

    const submitBtn = document.querySelector('#create-challenge-overlay button:last-child');
    if (submitBtn) { submitBtn.textContent = 'Creating…'; submitBtn.disabled = true; }

    try {
      const response = await fetch('/api/challenges/create', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          title, description: desc, sport,
          goal_type: goalType, goal_target: parseFloat(goalTarget),
          goal_unit: goalUnit, start_date: new Date(startDate).toISOString(),
          end_date: new Date(endDate).toISOString(),
          visibility, invitees
        })
      });
      const result = await response.json();
      if (result.error) {
        errorEl.textContent = 'Error: ' + result.error;
        errorEl.style.display = 'block';
        if (submitBtn) { submitBtn.textContent = 'Create challenge →'; submitBtn.disabled = false; }
        return;
      }
      document.getElementById('create-challenge-overlay')?.remove();
      showToast('✓ Challenge created! Invites sent to your followers.');
      setTimeout(() => loadChallenges(), 500);
    } catch (err) {
      errorEl.textContent = 'Something went wrong — please try again.';
      errorEl.style.display = 'block';
      if (submitBtn) { submitBtn.textContent = 'Create challenge →'; submitBtn.disabled = false; }
    }
  };

  // Wire Create challenge button
  const createChallengeBtn = [...document.querySelectorAll('button, .btn')]
    .find(b => b.textContent.includes('Create challenge'));
  if (createChallengeBtn) {
    createChallengeBtn.onclick = openCreateChallenge;
  }

  // Load on page open
  loadChallenges();
})();
