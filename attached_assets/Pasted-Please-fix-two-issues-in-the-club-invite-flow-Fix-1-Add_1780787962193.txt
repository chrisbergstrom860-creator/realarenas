function buildNotifActions(n) {
  if (n.type === 'club' && n.body?.toLowerCase().includes('invited you to join')) {
    // Extract club ID from the notification link

    const clubIdMatch = n.link?.match(/join-club\/([^?]+)/);

  }

}

javascriptwindow.acceptClubInvite = async function(clubId, notifId, btn) {
  btn.textContent = 'Joining…';
  btn.disabled = true;
  try {
    const response = await fetch(`/api/clubs/${clubId}/accept-invite`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' }
    });
    const result = await response.json();
    if (result.error) {
      showToast('Error: ' + result.error);
      btn.textContent = '✓ Accept invite';
      btn.disabled = false;
      return;
    }
    // Replace the action buttons with a success message
    const actionsEl = document.getElementById('invite-actions-' + notifId);
    if (actionsEl) {
      actionsEl.innerHTML = `
        <div style="font-size:12px;color:#166534;background:#ECFDF5;border:0.5px solid #86EFAC;padding:5px 12px;border-radius:20px;display:inline-flex;align-items:center;gap:5px">
          ✓ You joined the club! <a href="/feed" style="color:#166534;font-weight:600;text-decoration:underline">Go to feed →</a>
        </div>`;
    }
    showToast('✓ Welcome to the club! You can now see club content in your feed.');
    // Mark notification as read
    await fetch('/api/notifications/' + notifId + '/read', { method: 'POST' });
  } catch (err) {
    showToast('Something went wrong — please try again');
    btn.textContent = '✓ Accept invite';
    btn.disabled = false;
  }
};

window.declineClubInvite = async function(notifId, btn) {
  btn.textContent = 'Declining…';
  btn.disabled = true;
  try {
    // Mark notification as read and dismiss it
    await fetch('/api/notifications/' + notifId + '/read', { method: 'POST' });
    const actionsEl = document.getElementById('invite-actions-' + notifId);
    if (actionsEl) {
      actionsEl.innerHTML = `<div style="font-size:12px;color:var(--gray-400)">Invite declined</div>`;
    }
    showToast('Invite declined');
  } catch (err) {
    showToast('Something went wrong');
    btn.textContent = 'Decline';
    btn.disabled = false;
  }
};
