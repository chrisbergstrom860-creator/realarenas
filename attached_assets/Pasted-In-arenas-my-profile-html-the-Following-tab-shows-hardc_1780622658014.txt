// Handle unfollow from the following list
window.handleUnfollow = async function(userId, btn) {
  if (!confirm('Unfollow this athlete?')) return;
  try {
    const response = await fetch('/api/follow/' + userId, { method: 'DELETE' });
    const result = await response.json();
    if (result.error) { showToast('Error: ' + result.error); return; }
    const row = btn.closest('[style*="display:flex"]') || btn.parentElement;
    row.style.transition = 'opacity .25s';
    row.style.opacity = '0';
    setTimeout(() => row.remove(), 250);
    showToast('Unfollowed');
    // Update following count
    data.followingCount = Math.max(0, (data.followingCount || 1) - 1);
    document.querySelectorAll('.following-count').forEach(el => {
      el.textContent = data.followingCount;
    });
  } catch (err) {
    showToast('Something went wrong — please try again');
  }
};
