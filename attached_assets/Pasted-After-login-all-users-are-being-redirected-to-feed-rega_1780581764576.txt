After login, all users are being redirected to /feed regardless of their role. Please update the /auth/login route in server.js to check if the logged in user is a club admin or coach and redirect them to the correct page:
javascriptapp.post('/auth/login', async (req, res) => {
  const { email, password } = req.body;
  const { data, error } = await supabase.auth.signInWithPassword({
    email,
    password
  });
  if (error || !data?.session) {
    return res.redirect('/landing?error=invalid');
  }
  // Set session cookie
  res.cookie('sb-access-token', data.session.access_token, {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'lax',
    maxAge: 7 * 24 * 60 * 60 * 1000
  });
  try {
    // Check if this user is a club admin or coach
    const supabaseAdmin = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY
    );
    const { data: membership } = await supabaseAdmin
      .from('memberships')
      .select('role, club_id')
      .eq('user_id', data.user.id)
      .in('role', ['admin', 'coach'])
      .single();
    if (membership) {
      // Club admin or coach — send to dashboard
      return res.redirect('/clubs/dashboard');
    }
  } catch (err) {
    console.log('Role check error:', err.message);
  }
  // Regular athlete — send to feed
  res.redirect('/feed');
});
This means:

A user with role admin or coach in the memberships table → goes to /clubs/dashboard
A regular athlete with no admin/coach role → goes to /feed
Invalid credentials → back to /landing?error=invalid

Also update the / root route to do the same role check:
javascriptapp.get('/', async (req, res) => {
  const token = req.cookies['sb-access-token'];
  if (!token) return res.redirect('/landing');
  try {
    const { data: { user }, error } = await supabase.auth.getUser(token);
    if (error || !user) return res.redirect('/landing');
    const supabaseAdmin = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY
    );
    const { data: membership } = await supabaseAdmin
      .from('memberships')
      .select('role')
      .eq('user_id', user.id)
      .in('role', ['admin', 'coach'])
      .single();
    if (membership) return res.redirect('/clubs/dashboard');
    return res.redirect('/feed');
  } catch (err) {
    return res.redirect('/landing');
  }
});
