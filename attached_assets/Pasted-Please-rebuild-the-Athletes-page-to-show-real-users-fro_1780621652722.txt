javascriptapp.get('/athletes', requireAuth, async (req, res) => {
  try {
    const supabaseAdmin = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY
    );

    // Get logged in user's profile
    const { data: myProfile } = await supabaseAdmin
      .from('profiles')
      .select('name, handle, sports')
      .eq('id', req.user.id)
      .single();

    // Get all other users' profiles
    const { data: athletes } = await supabaseAdmin
      .from('profiles')
      .select('id, name, handle, sports, level, location, bio')
      .neq('id', req.user.id)
      .order('created_at', { ascending: false })
      .limit(50);

    // Get who the current user already follows
    const { data: following } = await supabaseAdmin
      .from('follows')
      .select('following_id')
      .eq('follower_id', req.user.id);

    const followingIds = (following || []).map(f => f.following_id);

    // Get post counts for each athlete
    const athleteIds = (athletes || []).map(a => a.id);
    const { data: postCounts } = await supabaseAdmin
      .from('posts')
      .select('user_id')
      .in('user_id', athleteIds);

    // Get follower counts
    const { data: followerCounts } = await supabaseAdmin
      .from('follows')
      .select('following_id')
      .in('following_id', athleteIds);

    // Enrich athletes with follow status and counts
    const enrichedAthletes = (athletes || []).map(athlete => {
      const posts = (postCounts || []).filter(p => p.user_id === athlete.id).length;
      const followers = (followerCounts || []).filter(f => f.following_id === athlete.id).length;
      const isFollowing = followingIds.includes(athlete.id);
      const initials = (athlete.name || 'A').split(' ').map(n => n[0]).join('').slice(0,2).toUpperCase();
      return {
        ...athlete,
        initials,
        postCount: posts,
        followerCount: followers,
        isFollowing
      };
    });

    const athleteData = {
      athletes: enrichedAthletes,
      myProfile: myProfile || { name: req.user.email, handle: 'athlete' },
      userId: req.user.id,
      followingIds
    };

    const fs = require('fs');
    let html = fs.readFileSync(path.join(HTML, 'arenas-athletes.html'), 'utf8');
    const dataScript = `<script>window.ARENAS_DATA = ${JSON.stringify(athleteData)};</script>`;
    html = html.replace('</head>', dataScript + '</head>');
    res.send(html);
  } catch (err) {
    console.log('Athletes error:', err.message);
    res.sendFile(path.join(HTML, 'arenas-athletes.html'));
  }
});

javascript(function() {

  // If no real athletes yet show empty state
  if (!data.athletes || data.athletes.length === 0) {
    const grid = document.getElementById('athlete-grid');
    if (grid) {
      grid.innerHTML = `
        <div style="grid-column:1/-1;text-align:center;padding:56px 24px;background:white">
          <div style="font-size:36px;margin-bottom:12px">👥</div>
          <h3 style="font-size:16px;font-weight:600;color:var(--gray-900);margin-bottom:6px">No athletes yet</h3>

        </div>`;
    }
    // Also hide the recommended strip and nearby section
    const recSection = document.getElementById('rec-section');
    const nearbySection = document.getElementById('nearby-section');
    if (recSection) recSection.style.display = 'none';
    if (nearbySection) nearbySection.style.display = 'none';
    return;
  }

  // Hide the hardcoded recommended strip and nearby section
  // since we don't have enough data for smart recommendations yet
  const recSection = document.getElementById('rec-section');
  const nearbySection = document.getElementById('nearby-section');
  if (recSection) recSection.style.display = 'none';
  if (nearbySection) nearbySection.style.display = 'none';

  // Sport color and emoji mapping
  const sportColors = {
    running: 'background:var(--orange-light);color:#9A3412;border-color:#FDBA74',
    cycling: 'background:var(--blue-light);color:#1E40AF;border-color:#93C5FD',
    climbing: 'background:var(--purple-light);color:#5B21B6;border-color:#C4B5FD',
    swimming: 'background:var(--teal-light);color:#134E4A;border-color:#5EEAD4',
    triathlon: 'background:var(--pink-light);color:#72243E;border-color:#F4C0D1',
    football: 'background:var(--green-light);color:#166534;border-color:#86EFAC'

  const sportEmojis = {
    running: '🏃 Running',
    cycling: '🚴 Cycling',
    climbing: '🧗 Climbing',
    swimming: '🏊 Swimming',

    football: '⚽ Football'

  // Avatar background colors — cycle through a set
  const avColors = [

    {bg:'#FCE7F3',color:'#9D174D'},
    {bg:'#E0F2FE',color:'#0369A1'},
    {bg:'#F5F3FF',color:'#5B21B6'},
    {bg:'#ECFDF5',color:'#065F46'},
    {bg:'#FFF7ED',color:'#9A3412'},
    {bg:'#F0FDFA',color:'#134E4A'},
    {bg:'#FBEAF0',color:'#72243E'}
  ];

  // Render real athlete cards
  const grid = document.getElementById('athlete-grid');
  if (!grid) return;

  const cardsHTML = data.athletes.map((athlete, i) => {
    const av = avColors[i % avColors.length];
    const primarySport = athlete.sports?.[0] || null;
    const sportStyle = primarySport ? sportColors[primarySport] || '' : '';
    const sportLabel = primarySport ? (sportEmojis[primarySport] || primarySport) : null;
    const levelLabel = athlete.level ?

    return `

  // Handle real follow/unfollow
  window.handleFollow = async function(userId, btn) {
    const isFollowing = btn.classList.contains('following');

      const method = isFollowing ? 'DELETE' : 'POST';

      const result = await response.json();

        btn.classList.remove('default');

        btn.textContent = 'Following';
        showToast('✓ Following — their posts will appear in your feed');
      } else {
        btn.classList.remove('following');
        btn.classList.add('default');
        btn.textContent = 'Follow';
        showToast('Unfollowed');

      btn.disabled = false;

})();
