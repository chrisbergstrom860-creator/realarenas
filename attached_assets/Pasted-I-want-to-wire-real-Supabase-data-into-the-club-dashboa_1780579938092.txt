Update the /clubs/dashboard route in server.js to fetch real data before serving the page:

javascriptapp.get('/clubs/dashboard', requireAuth, async (req, res) => {
  try {
    const supabaseAdmin = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY
    );

    // Get the club this user administers
    const { data: membership } = await supabaseAdmin
      .from('memberships')
      .select('club_id, role, clubs(id, name, handle, sport, city, plan)')
      .eq('user_id', req.user.id)
      .in('role', ['admin', 'coach'])
      .single();

    // Get the user's profile
    const { data: profile } = await supabaseAdmin
      .from('profiles')
      .select('name, handle')
      .eq('id', req.user.id)
      .single();

    // Get member count for this club
    const { count: memberCount } = await supabaseAdmin
      .from('memberships')
      .select('*', { count: 'exact', head: true })
      .eq('club_id', membership?.club_id)
      .eq('status', 'active');

    // Get recent members
    const { data: members } = await supabaseAdmin
      .from('memberships')
      .select('user_id, role, joined_at, profiles(name, handle)')
      .eq('club_id', membership?.club_id)
      .eq('status', 'active')
      .order('joined_at', { ascending: false })
      .limit(10);

    // Build the data object to inject into the page
    const clubData = {
      club: membership?.clubs || null,
      profile: profile || { name: req.user.email, handle: 'coach' },
      memberCount: memberCount || 0,
      members: members || [],
      userEmail: req.user.email
    };

    // Read the HTML file
    const fs = require('fs');
    const path = require('path');
    let html = fs.readFileSync(path.join(HTML, 'arenas-club-dashboard.html'), 'utf8');

    // Inject the real data as a JavaScript variable before the closing </head> tag
    const dataScript = `<script>window.ARENAS_DATA = ${JSON.stringify(clubData)};</script>`;
    html = html.replace('</head>', dataScript + '</head>');

    res.send(html);
  } catch (err) {
    console.log('Dashboard data error:', err.message);
    res.sendFile(path.join(HTML, 'arenas-club-dashboard.html'));
  }
});

Add this JavaScript at the bottom of arenas-club-dashboard.html before the closing </body> tag to replace hardcoded values with real data:

javascript// Wire real data from server
(function() {
  const data = window.ARENAS_DATA;
  if (!data) return;

  // Replace club name everywhere it appears
  const clubName = data.club?.name || 'Your Club';
  const coachName = data.profile?.name || 'Coach';
  const initials = coachName.split(' ').map(n => n[0]).join('').slice(0,2).toUpperCase();

  // Update all elements with class .club-name
  document.querySelectorAll('.club-name').forEach(el => el.textContent = clubName);

  // Update all elements with class .coach-name
  document.querySelectorAll('.coach-name').forEach(el => el.textContent = coachName);

  // Update all avatar initials with class .coach-initials
  document.querySelectorAll('.coach-initials').forEach(el => el.textContent = initials);

  // Update member count
  document.querySelectorAll('.member-count').forEach(el => el.textContent = data.memberCount);

  // Update page title
  document.title = 'Arenas · ' + clubName + ' Dashboard';

  // Update club sport tag if it exists
  if (data.club?.sport) {
    document.querySelectorAll('.club-sport').forEach(el => {
      el.textContent = data.club.sport[0] || 'Multi-sport';
    });
  }

  // Update the plan badge
  if (data.club?.plan) {
    document.querySelectorAll('.club-plan').forEach(el => {
      el.textContent = data.club.plan.charAt(0).toUpperCase() + data.club.plan.slice(1) + ' plan';
    });
  }
})();
