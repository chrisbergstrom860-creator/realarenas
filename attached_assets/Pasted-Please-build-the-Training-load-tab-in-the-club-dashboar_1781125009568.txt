javascript// Parse duration from activity — handles "45", "45 min", "1h 30m", "1:30" formats
function parseDurationHours(duration) {
  if (!duration) return 0;
  const str = String(duration).toLowerCase().trim();
  // "1:30" format
  if (str.includes(':')) {
    const parts = str.split(':');
    const h = parseFloat(parts[0]) || 0;
    const m = parseFloat(parts[1]) || 0;
    return h + m / 60;
  }
  // "1h 30m" format
  const hMatch = str.match(/(\d+(?:\.\d+)?)\s*h/);
  const mMatch = str.match(/(\d+(?:\.\d+)?)\s*m/);
  if (hMatch || mMatch) {
    return (parseFloat(hMatch?.[1]) || 0) + (parseFloat(mMatch?.[1]) || 0) / 60;
  }
  // Plain number — assume minutes
  const num = parseFloat(str.replace(/[^0-9.]/g, ''));
  if (isNaN(num)) return 0;
  return num > 12 ? num / 60 : num; // >12 means minutes, otherwise hours
}

// Get the Monday of a week offset from now (0 = this week)
function getWeekStart(weeksAgo) {
  const now = new Date();
  const day = now.getDay() || 7;
  const monday = new Date(now);
  monday.setDate(now.getDate() - day + 1 - weeksAgo * 7);
  monday.setHours(0, 0, 0, 0);
  return monday;
}

app.get('/api/clubs/:clubId/training-load', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const weeks = parseInt(req.query.weeks) || 6;
  // Check requester is admin/coach of this club
  const { data: membership } = await supabaseAdmin
    .from('memberships')
    .select('role')
    .eq('user_id', req.user.id)
    .eq('club_id', req.params.clubId)
    .in('role', ['admin', 'coach'])
    .single();
  if (!membership) return res.json({ error: 'Not authorised' });
  // Get all club members
  const { data: members } = await supabaseAdmin
    .from('memberships')
    .select('user_id, profiles:user_id(id, name, handle, sports)')
    .eq('club_id', req.params.clubId);
  const memberIds = (members || []).map(m => m.user_id);
  // Get all activities in the period
  const periodStart = getWeekStart(weeks - 1);
  const { data: activities } = await supabaseAdmin
    .from('activities')
    .select('user_id, sport, distance, duration, date')
    .in('user_id', memberIds.length ? memberIds : ['00000000-0000-0000-0000-000000000000'])
    .gte('date', periodStart.toISOString());
  // Build week boundaries (oldest first)
  const weekStarts = [];
  for (let i = weeks - 1; i >= 0; i--) {
    weekStarts.push(getWeekStart(i));
  }
  // Calculate per-member per-week hours
  const memberData = (members || []).map(m => {
    const memberActivities = (activities || []).filter(a => a.user_id === m.user_id);
    const weeklyHours = weekStarts.map((weekStart, i) => {
      const weekEnd = new Date(weekStart);
      weekEnd.setDate(weekEnd.getDate() + 7);
      const weekActs = memberActivities.filter(a => {
        const d = new Date(a.date);
        return d >= weekStart && d < weekEnd;
      });
      return Math.round(weekActs.reduce((sum, a) => sum + parseDurationHours(a.duration), 0) * 10) / 10;
    });
    const thisWeek = weeklyHours[weeklyHours.length - 1];
    // 4-week average excluding current week
    const prevWeeks = weeklyHours.slice(Math.max(0, weeklyHours.length - 5), weeklyHours.length - 1);
    const avg = prevWeeks.length > 0
      ? Math.round((prevWeeks.reduce((s, h) => s + h, 0) / prevWeeks.length) * 10) / 10
      : 0;
    // This week stats
    const thisWeekStart = weekStarts[weekStarts.length - 1];
    const thisWeekActs = memberActivities.filter(a => new Date(a.date) >= thisWeekStart);
    const thisWeekKm = Math.round(thisWeekActs.reduce((sum, a) => {
      const dist = parseFloat((a.distance || '0').replace(/[^0-9.]/g, ''));
      return sum + (isNaN(dist) ? 0 : dist);
    }, 0) * 10) / 10;
    const activeDays = new Set(thisWeekActs.map(a => new Date(a.date).toDateString())).size;
    const restDays = Math.max(0, Math.min(7, new Date().getDay() || 7) - activeDays);
    // Status calculation
    let status, trend;
    if (thisWeek === 0 && avg === 0) {
      status = 'inactive'; trend = 0;
    } else if (thisWeek === 0) {
      status = 'inactive'; trend = -100;
    } else if (avg === 0) {
      status = 'ontrack'; trend = 0;
    } else {
      trend = Math.round(((thisWeek - avg) / avg) * 100);
      if (trend >= 50) status = 'overdoing';
      else if (trend <= -40) status = 'behind';
      else status = 'ontrack';
    }
    return {
      userId: m.user_id,
      name: m.profiles?.name || 'Member',
      handle: m.profiles?.handle || 'member',
      sports: m.profiles?.sports || [],
      weeklyHours,
      thisWeek,
      avg,
      trend,
      status,
      sessionsThisWeek: thisWeekActs.length,
      kmThisWeek: thisWeekKm,
      restDays
    };
  });
  // Sort: overdoing first, then behind, then ontrack, then inactive
  const statusOrder = { overdoing: 0, behind: 1, ontrack: 2, inactive: 3 };
  memberData.sort((a, b) => statusOrder[a.status] - statusOrder[b.status] || b.thisWeek - a.thisWeek);
  // Club totals per week
  const clubWeekly = weekStarts.map((_, i) =>
    Math.round(memberData.reduce((sum, m) => sum + (m.weeklyHours[i] || 0), 0) * 10) / 10
  );
  const clubThisWeek = clubWeekly[clubWeekly.length - 1];
  const clubPrev = clubWeekly.slice(Math.max(0, clubWeekly.length - 5), clubWeekly.length - 1);
  const clubAvg = clubPrev.length > 0
    ? Math.round((clubPrev.reduce((s, h) => s + h, 0) / clubPrev.length) * 10) / 10
    : 0;
  res.json({
    members: memberData,
    clubWeekly,
    weekLabels: weekStarts.map(d => d.toLocaleDateString('en-GB', { day: 'numeric', month: 'short' })),
    stats: {
      clubThisWeek,
      clubAvg,
      clubDelta: Math.round((clubThisWeek - clubAvg) * 10) / 10,
      activeCount: memberData.filter(m => m.thisWeek > 0).length,
      totalMembers: memberData.length,
      overdoingCount: memberData.filter(m => m.status === 'overdoing').length,
      behindCount: memberData.filter(m => m.status === 'behind').length
    }
  });
});

async function loadTrainingLoad() {
  const data = window.ARENAS_DATA;
  const clubId = data?.club?.id;
  if (!clubId) return;
  try {
    const response = await fetch(`/api/clubs/${clubId}/training-load?weeks=${tlWeeks}`);
    const result = await response.json();
    if (result.error) return;
    tlMembers = result.members || [];

    // Stats
    const s = result.stats || {};
    const setEl = (id, val) => { const el = document.getElementById(id); if (el) el.textContent = val; };
    setEl('tl-stat-hours', (s.clubThisWeek || 0) + 'h');
    setEl('tl-stat-active', s.activeCount || 0);
    setEl('tl-stat-total', 'of ' + (s.totalMembers || 0) + ' total');
    setEl('tl-stat-over', s.overdoingCount || 0);
    setEl('tl-stat-behind', s.behindCount || 0);
    const deltaEl = document.getElementById('tl-stat-delta');
    if (deltaEl) {
      const d = s.clubDelta || 0;
      deltaEl.textContent = d >= 0 ? `↑ +${d}h vs 4-week avg` : `↓ ${d}h vs 4-week avg`;
      deltaEl.style.color = d >= 0 ? '#10B981' : '#854D0E';
    }

    // Club chart
    const chartEl = document.getElementById('tl-chart-bars');
    if (chartEl) {
      const weekly = result.clubWeekly || [];
      const labels = result.weekLabels || [];
      const maxVal = Math.max(...weekly, 1);
      chartEl.innerHTML = weekly.map((hours, i) => {
        const isCurrent = i === weekly.length - 1;
        const heightPct = Math.max(4, Math.round((hours / maxVal) * 88));
        return `
          <div style="flex:1;display:flex;flex-direction:column;align-items:center;gap:5px;height:100%;justify-content:flex-end">
            <div style="font-size:10px;font-family:var(--mono);color:${isCurrent ? 'var(--gray-900)' : 'var(--gray-500)'};font-weight:${isCurrent ? '600' : '400'}">${hours}h</div>
            <div style="width:100%;max-width:44px;height:${heightPct}%;border-radius:4px 4px 0 0;background:${isCurrent ? '#E6B800' : '#FFD21E'}"></div>
            <div style="font-size:10px;color:${isCurrent ? 'var(--gray-600)' : 'var(--gray-400)'};font-weight:${isCurrent ? '600' : '400'}">${isCurrent ? 'This week' : labels[i]}</div>
          </div>`;
      }).join('');
    }

    renderTlMembers();
  } catch (err) {
    console.log('Could not load training load:', err);
  }
}

window.sendTlCheckin = async function(userId, name, btn) {
  const data = window.ARENAS_DATA;
  const clubId = data?.club?.id;
  if (!clubId) return;
  btn.textContent = 'Sending…';
  btn.disabled = true;
  try {
    const response = await fetch(`/api/clubs/${clubId}/nudge-atrisk`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ userIds: [userId] })
    });
    const result = await response.json();
    if (result.error) { showToast('Error: ' + result.error); btn.textContent = 'Send check-in'; btn.disabled = false; return; }
    showToast(`✓ Check-in sent to ${name}`);
    btn.textContent = '✓ Sent';
  } catch (err) {
    showToast('Something went wrong');
    btn.textContent = 'Send check-in';
    btn.disabled = false;
  }
};
