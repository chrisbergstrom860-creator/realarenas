Update the /feed route in server.js to also fetch real posts and inject them into the page alongside the user profile data:

javascriptapp.get('/feed', requireAuth, async (req, res) => {
  try {
    const supabaseAdmin = createClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY
    );

    // Get logged in user's profile
    const { data: profile } = await supabaseAdmin
      .from('profiles')
      .select('name, handle, sports, level, location')
      .eq('id', req.user.id)
      .single();

    // Get real posts from people this user follows
    // plus their own posts
    const { data: posts } = await supabaseAdmin
      .from('posts')
      .select(`
        id,
        content,
        sport,
        feeling,
        created_at,
        user_id
      `)
      .order('created_at', { ascending: false })
      .limit(20);

    // Get like counts for each post
    const postIds = (posts || []).map(p => p.id);
    const { data: likes } = await supabaseAdmin
      .from('post_likes')
      .select('post_id, user_id')
      .in('post_id', postIds);

    // Get comment counts
    const { data: comments } = await supabaseAdmin
      .from('post_comments')
      .select('post_id')
      .in('post_id', postIds);

    // Get profiles for post authors
    const authorIds = [...new Set((posts || []).map(p => p.user_id))];
    const { data: authorProfiles } = await supabaseAdmin
      .from('profiles')
      .select('id, name, handle')
      .in('id', authorIds);

    // Build enriched posts array
    const enrichedPosts = (posts || []).map(post => {
      const author = (authorProfiles || []).find(p => p.id === post.user_id);
      const postLikes = (likes || []).filter(l => l.post_id === post.id);
      const postComments = (comments || []).filter(c => c.post_id === post.id);
      const userLiked = postLikes.some(l => l.user_id === req.user.id);
      return {
        ...post,
        authorName: author?.name || 'Athlete',
        authorHandle: author?.handle || 'athlete',
        likeCount: postLikes.length,
        commentCount: postComments.length,
        userLiked
      };
    });

    const userData = {
      profile: profile || { name: req.user.email, handle: 'athlete' },
      userId: req.user.id,
      posts: enrichedPosts
    };

    const fs = require('fs');
    let html = fs.readFileSync(path.join(HTML, 'arenas-feed.html'), 'utf8');
    const dataScript = `<script>window.ARENAS_DATA = ${JSON.stringify(userData)};</script>`;
    html = html.replace('</head>', dataScript + '</head>');
    res.send(html);
  } catch (err) {
    console.log('Feed error:', err.message);
    res.sendFile(path.join(HTML, 'arenas-feed.html'));
  }
});

Update arenas-feed.html — add this JavaScript at the bottom before </body> to render real posts into the feed. Find the element that contains the feed cards — it probably has an id like feed-items or similar — and prepend real posts to it:

javascript(function() {
  const data = window.ARENAS_DATA;
  if (!data) return;

  // Update user name and initials throughout the page
  const name = data.profile?.name || 'Athlete';
  const initials = name.split(' ').map(n => n[0]).join('').slice(0,2).toUpperCase();
  const handle = data.profile?.handle || 'athlete';

  document.querySelectorAll('.user-name').forEach(el => el.textContent = name);
  document.querySelectorAll('.user-handle').forEach(el => el.textContent = '@' + handle);
  document.querySelectorAll('.user-av, .composer-av').forEach(el => {
    if (el.textContent.trim() === 'JK' || el.textContent.trim() === '') {
      el.textContent = initials;
    }
  });

  // Render real posts at the top of the feed
  if (!data.posts || data.posts.length === 0) return;

  const feedContainer = document.getElementById('feed-items');
  if (!feedContainer) return;

  const sportColors = {
    running: 'background:#FFF7ED;color:#9A3412;border:1px solid #FDBA74',
    cycling: 'background:#EFF6FF;color:#1E40AF;border:1px solid #93C5FD',
    climbing: 'background:#F5F3FF;color:#5B21B6;border:1px solid #C4B5FD',
    swimming: 'background:#F0FDFA;color:#134E4A;border:1px solid #5EEAD4'
  };
  const sportEmojis = {
    running: '🏃 Running',
    cycling: '🚴 Cycling',
    climbing: '🧗 Climbing',
    swimming: '🏊 Swimming'
  };
  const feelingLabels = {
    great: '💪 Feeling great',
    tired: '😴 Tired',
    injured: '🤕 Injured',
    motivated: '🔥 Motivated'
  };

  function timeAgo(dateStr) {
    const seconds = Math.floor((new Date() - new Date(dateStr)) / 1000);
    if (seconds < 60) return 'Just now';
    if (seconds < 3600) return Math.floor(seconds/60) + 'm ago';
    if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
    if (seconds < 604800) return Math.floor(seconds/86400) + 'd ago';
    return new Date(dateStr).toLocaleDateString();
  }

  function getInitials(name) {
    return (name || 'A').split(' ').map(n => n[0]).join('').slice(0,2).toUpperCase();
  }

  // Clear existing hardcoded post note cards first
  document.querySelectorAll('.post-note-card').forEach(card => {
    card.closest('.feed-item-wrap')?.remove();
  });

  // Render real posts
  const postsHTML = data.posts.map(post => {
    const postInitials = getInitials(post.authorName);
    const isOwn = post.user_id === data.userId;
    return `
      <div class="feed-item-wrap">
        <div class="post-note-card">
          <div class="pn-head">
            <div class="pn-av" style="background:${isOwn ? '#FFD21E' : '#F3F4F6'};color:${isOwn ? '#111827' : '#374151'}">${postInitials}</div>
            <div>
              <div class="pn-name">${post.authorName}${isOwn ? ' <span style="font-size:11px;color:#9CA3AF;font-weight:400">(you)</span>' : ''}</div>
              <div class="pn-meta">${timeAgo(post.created_at)} · @${post.authorHandle}</div>
            </div>
            ${post.sport ? `<div class="pn-sport-tag" style="${sportColors[post.sport] || ''}">${sportEmojis[post.sport] || post.sport}</div>` : ''}
          </div>
          ${post.feeling ? `<div class="pn-feeling">${feelingLabels[post.feeling] || post.feeling}</div>` : ''}
          <div class="pn-content">${post.content}</div>
          <div class="pn-footer">
            <button class="pn-action ${post.userLiked ? 'liked' : ''}" onclick="likePost(this,'${post.id}')">
              👍 ${post.likeCount} ${post.likeCount === 1 ? 'kudo' : 'kudos'}
            </button>
            <button class="pn-action">💬 ${post.commentCount} ${post.commentCount === 1 ? 'comment' : 'comments'}</button>
            <button class="pn-action" onclick="showToast('Link copied!')">🔗 Share</button>
            <span style="margin-left:auto;font-size:11px;color:#9CA3AF">Training note</span>
          </div>
        </div>
      </div>`;
  }).join('');

  feedContainer.insertAdjacentHTML('afterbegin', postsHTML);
})();
