javascriptapp.post('/api/clubs/:clubId/invite', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { email, role } = req.body;
  const { clubId } = req.params;
  if (!email || !email.includes('@')) {
    return res.json({ error: 'Please enter a valid email address' });
  }
  // Check the requester is an admin or coach of this club
  const { data: membership } = await supabaseAdmin
    .from('memberships')
    .select('role')
    .eq('user_id', req.user.id)
    .eq('club_id', clubId)
    .in('role', ['admin', 'coach'])
    .single();
  if (!membership) return res.json({ error: 'Not authorised' });
  // Get club and inviter details
  const { data: club } = await supabaseAdmin
    .from('clubs')
    .select('name, handle, sport')
    .eq('id', clubId)
    .single();
  const { data: inviterProfile } = await supabaseAdmin
    .from('profiles')
    .select('name')
    .eq('id', req.user.id)
    .single();
  // Check if this email already belongs to an Arenas user
  const { data: existingUsers } = await supabaseAdmin
    .auth.admin.listUsers();
  const existingUser = (existingUsers?.users || [])
    .find(u => u.email?.toLowerCase() === email.toLowerCase());
  if (existingUser) {
    // User already has an account — check if already a member
    const { data: existingMembership } = await supabaseAdmin
      .from('memberships')
      .select('id')
      .eq('user_id', existingUser.id)
      .eq('club_id', clubId)
      .single();
    if (existingMembership) {
      return res.json({ error: 'This user is already a member of your club' });
    }
    // Send in-app notification to the existing user
    await createNotification(supabaseAdmin, {
      userId: existingUser.id,
      type: 'club',
      title: 'Club invite',
      body: `${inviterProfile?.name || 'A coach'} invited you to join ${club?.name} on Arenas`,
      link: `/join-club/${clubId}?token=${generateInviteToken()}`,
      actorId: req.user.id,
      entityId: clubId
    });
    // Also create a club_invite record so they can accept it
    const token = generateInviteToken();
    await supabaseAdmin.from('club_invites').insert({
      club_id: clubId,
      invited_by: req.user.id,
      email: email.toLowerCase(),
      role: role || 'member',
      token,
      status: 'pending',
      expires_at: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString()
    });
    const baseUrl = process.env.NODE_ENV === 'production'
      ? 'https://realarenas.com'
      : `http://localhost:${process.env.PORT || 3000}`;
    return res.json({
      success: true,
      existingUser: true,
      message: `${email} already has an Arenas account — they've been notified in-app and can join with one click`,
      joinUrl: `${baseUrl}/join/${token}`
    });
  }
  // New user — create invite with join link as before
  // Check if already invited
  const { data: existingInvite } = await supabaseAdmin
    .from('club_invites')
    .select('id, status')
    .eq('club_id', clubId)
    .eq('email', email.toLowerCase())
    .eq('status', 'pending')
    .single();
  if (existingInvite) {
    return res.json({ error: 'This email already has a pending invite' });
  }
  const token = generateInviteToken();
  const { data: invite, error } = await supabaseAdmin
    .from('club_invites')
    .insert({
      club_id: clubId,
      invited_by: req.user.id,
      email: email.toLowerCase(),
      role: role || 'member',
      token,
      status: 'pending',
      expires_at: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString()
    })
    .select()
    .single();
  if (error) return res.json({ error: error.message });
  const baseUrl = process.env.NODE_ENV === 'production'
    ? 'https://realarenas.com'
    : `http://localhost:${process.env.PORT || 3000}`;
  const joinUrl = `${baseUrl}/join/${token}`;
  console.log('INVITE EMAIL - New user:');
  console.log('To:', email);
  console.log('Join URL:', joinUrl);
  res.json({
    success: true,
    existingUser: false,
    message: `Invite created for ${email}`,
    joinUrl
  });
});

javascriptapp.post('/api/clubs/:clubId/accept-invite', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { clubId } = req.params;
  // Check not already a member
  const { data: existing } = await supabaseAdmin
    .from('memberships')
    .select('id')
    .eq('user_id', req.user.id)
    .eq('club_id', clubId)
    .single();
  if (existing) return res.json({ error: 'You are already a member of this club' });
  // Find their pending invite
  const { data: userEmail } = await supabaseAdmin
    .auth.admin.getUserById(req.user.id);
  const { data: invite } = await supabaseAdmin
    .from('club_invites')
    .select('*')
    .eq('club_id', clubId)
    .eq('email', userEmail?.user?.email?.toLowerCase() || '')
    .eq('status', 'pending')
    .single();
  // Add membership
  const { error } = await supabaseAdmin
    .from('memberships')
    .insert({
      user_id: req.user.id,
      club_id: clubId,
      role: invite?.role || 'member',
      status: 'active'
    });
  if (error && !error.message.includes('duplicate')) {
    return res.json({ error: error.message });
  }
  // Mark invite accepted
  if (invite) {
    await supabaseAdmin
      .from('club_invites')
      .update({ status: 'accepted', accepted_at: new Date().toISOString() })
      .eq('id', invite.id);
  }
  // Notify the club admin
  const { data: profile } = await supabaseAdmin
    .from('profiles')
    .select('name')
    .eq('id', req.user.id)
    .single();
  const { data: club } = await supabaseAdmin
    .from('clubs')
    .select('name')
    .eq('id', clubId)
    .single();
  const { data: admins } = await supabaseAdmin
    .from('memberships')
    .select('user_id')
    .eq('club_id', clubId)
    .eq('role', 'admin');
  for (const admin of (admins || [])) {
    if (admin.user_id !== req.user.id) {
      await createNotification(supabaseAdmin, {
        userId: admin.user_id,
        type: 'club',
        title: 'New member joined',
        body: `${profile?.name || 'Someone'} accepted your invite and joined ${club?.name}`,
        link: '/clubs/dashboard',
        actorId: req.user.id,
        entityId: clubId
      });
    }
  }
  res.json({ success: true, message: 'Welcome to the club!' });
});
