2. Trigger a notification when someone likes a post — update the /api/posts/:id/like route to create a notification for the post author when someone likes their post:
javascriptapp.post('/api/posts/:id/like', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data: existing } = await supabaseAdmin
    .from('post_likes')
    .select()
    .eq('post_id', req.params.id)
    .eq('user_id', req.user.id)
    .single();
  if (existing) {
    await supabaseAdmin.from('post_likes').delete()
      .eq('post_id', req.params.id)
      .eq('user_id', req.user.id);
    return res.json({ liked: false });
  }
  await supabaseAdmin.from('post_likes').insert({
    post_id: req.params.id,
    user_id: req.user.id
  });
  // Get the post to find the author
  const { data: post } = await supabaseAdmin
    .from('posts')
    .select('user_id, content')
    .eq('id', req.params.id)
    .single();
  // Get the liker's profile
  const { data: likerProfile } = await supabaseAdmin
    .from('profiles')
    .select('name, handle')
    .eq('id', req.user.id)
    .single();
  // Only notify if someone else liked it — not yourself
  if (post && post.user_id !== req.user.id) {
    await createNotification(supabaseAdmin, {
      userId: post.user_id,
      type: 'like',
      title: 'New kudos',
      body: `${likerProfile?.name || 'Someone'} gave kudos on your training note: "${post.content.slice(0, 60)}${post.content.length > 60 ? '...' : ''}"`,
      link: '/feed',
      actorId: req.user.id,
      entityId: req.params.id
    });
  }
  res.json({ liked: true });
});

  // Get the post author
  const { data: post } = await supabaseAdmin
    .from('posts')

    .single();
  // Get commenter profile
  const { data: commenterProfile } = await supabaseAdmin
    .from('profiles')
    .select('name, handle')
    .eq('id', req.user.id)
    .single();
  // Notify post author if someone else commented
  if (post && post.user_id !== req.user.id) {
    await createNotification(supabaseAdmin, {
      userId: post.user_id,
      type: 'comment',
      title: 'New comment',

javascript// Get notifications for logged in user
app.get('/api/notifications', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data, error } = await supabaseAdmin
    .from('notifications')
    .select('*, actor:actor_id(name, handle)')
    .eq('user_id', req.user.id)
    .order('created_at', { ascending: false })
    .limit(50);
  if (error) return res.json({ error: error.message });
  const unreadCount = (data || []).filter(n => !n.read).length;
  res.json({ notifications: data || [], unreadCount });
});

    if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
