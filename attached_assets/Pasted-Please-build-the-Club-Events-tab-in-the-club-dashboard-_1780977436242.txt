app.post('/api/events/:id/nudge', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  // Get the event
  const { data: event } = await supabaseAdmin
    .from('events')
    .select('*, clubs:club_id(name)')
    .eq('id', req.params.id)
    .single();
  if (!event) return res.json({ error: 'Event not found' });
  // Get all club members
  const { data: members } = await supabaseAdmin
    .from('memberships')
    .select('user_id')
    .eq('club_id', event.club_id);
  // Get members who already responded
  const { data: rsvps } = await supabaseAdmin
    .from('event_rsvps')
    .select('user_id')
    .eq('event_id', req.params.id);
  const respondedIds = new Set((rsvps || []).map(r => r.user_id));
  // Get coach profile
  const { data: profile } = await supabaseAdmin
    .from('profiles')
    .select('name')
    .eq('id', req.user.id)
    .single();
  // Notify only non-responders
  const nonResponders = (members || []).filter(m => !respondedIds.has(m.user_id) && m.user_id !== req.user.id);
  const eventDate = new Date(event.date).toLocaleDateString('en-GB', {
    weekday: 'long', day: 'numeric', month: 'long'
  });
  for (const member of nonResponders) {
    await createNotification(supabaseAdmin, {
      userId: member.user_id,
      type: 'club',
      title: 'RSVP reminder',
      body: `${profile?.name || 'Your coach'} is asking — are you coming to "${event.title}" on ${eventDate}? Please RSVP so they can plan ahead.`,
      link: '/events',
      actorId: req.user.id,
      entityId: event.id
    });
  }
  res.json({ success: true, nudged: nonResponders.length });
});

// Duplicate an event
app.post('/api/events/:id/duplicate', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { data: event } = await supabaseAdmin
    .from('events')
    .select('*')
    .eq('id', req.params.id)
    .single();
  if (!event) return res.json({ error: 'Event not found' });
  // Create duplicate with date one week later
  const newDate = new Date(event.date);
  newDate.setDate(newDate.getDate() + 7);
  const { data: newEvent, error } = await supabaseAdmin
    .from('events')
    .insert({
      created_by: req.user.id,
      club_id: event.club_id,
      title: event.title,
      sport: event.sport,
      event_type: event.event_type,
      date: newDate.toISOString(),
      location: event.location,
      distance: event.distance,
      max_participants: event.max_participants,
      entry_fee: event.entry_fee,
      level: event.level,
      description: event.description,
      visibility: event.visibility
    })
    .select()
    .single();
  if (error) return res.json({ error: error.message });
  res.json({ success: true, event: newEvent });
});

  window.submitCreateClubEvent = async function(clubId) {
    const title = document.getElementById('cev-title')?.value?.trim();
    const sport = document.querySelector('.cev-sport-chip.selected')?.dataset?.sport || 'running';
    const event_type = document.getElementById('cev-type')?.value;
    const date = document.getElementById('cev-date')?.value;
    const time = document.getElementById('cev-time')?.value || '07:00';
    const location = document.getElementById('cev-location')?.value?.trim();
    const distance = document.getElementById('cev-distance')?.value?.trim();
    const level = document.getElementById('cev-level')?.value;
    const description = document.getElementById('cev-desc')?.value?.trim();
    const errorEl = document.getElementById('cev-error');
    if (!title) { errorEl.textContent = 'Please enter an event title.'; errorEl.style.display = 'block'; return; }
    if (!date) { errorEl.textContent = 'Please select a date.'; errorEl.style.display = 'block'; return; }
    if (!location) { errorEl.textContent = 'Please enter a location.'; errorEl.style.display = 'block'; return; }
    const submitBtn = document.querySelector('#create-club-event-overlay button:last-child');
    if (submitBtn) { submitBtn.textContent = 'Creating…'; submitBtn.disabled = true; }
    try {
      const response = await fetch('/api/events/create', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          title, sport, event_type,
          date: new Date(date + 'T' + time).toISOString(),
          location, distance, level, description,
          visibility: 'club',
          club_id: clubId,
          invitees: []
        })
      });
      const result = await response.json();
      if (result.error) {
        errorEl.textContent = 'Error: ' + result.error;
        errorEl.style.display = 'block';
        if (submitBtn) { submitBtn.textContent = 'Create & notify members →'; submitBtn.disabled = false; }
        return;
      }
      document.getElementById('create-club-event-overlay')?.remove();

      setTimeout(() => window.location.reload(), 1500);
    } catch (err) {
      errorEl.textContent = 'Something went wrong — please try again.';
      errorEl.style.display = 'block';
      if (submitBtn) { submitBtn.textContent = 'Create & notify members →'; submitBtn.disabled = false; }
    }
