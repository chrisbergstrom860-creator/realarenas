app.post('/api/clubs/:clubId/invite/bulk', requireAuth, async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { emails, role } = req.body;
  const { clubId } = req.params;
  if (!emails || !Array.isArray(emails) || emails.length === 0) {
    return res.json({ error: 'No valid emails provided' });
  }
  // Check authorisation
  const { data: membership } = await supabaseAdmin
    .from('memberships')
    .select('role')
    .eq('user_id', req.user.id)
    .eq('club_id', clubId)
    .in('role', ['admin', 'coach'])
    .single();
  if (!membership) return res.json({ error: 'Not authorised' });
  const results = [];
  const baseUrl = process.env.NODE_ENV === 'production'
    ? 'https://realarenas.com'
    : `http://localhost:${process.env.PORT || 3000}`;
  for (const email of emails) {
    const cleanEmail = email.toLowerCase().trim();
    if (!cleanEmail.includes('@')) continue;
    try {
      const token = generateInviteToken();
      const { data: invite } = await supabaseAdmin
        .from('club_invites')
        .insert({
          club_id: clubId,
          invited_by: req.user.id,
          email: cleanEmail,
          role: role || 'member',
          token,
          status: 'pending',
          expires_at: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString()
        })
        .select()
        .single();
      if (invite) {
        results.push({ email: cleanEmail, joinUrl: `${baseUrl}/join/${token}`, success: true });
        console.log('BULK INVITE - To:', cleanEmail, 'Join URL:', `${baseUrl}/join/${token}`);
      }
    } catch (err) {
      results.push({ email: cleanEmail, success: false, error: err.message });
    }
  }
  res.json({ success: true, sent: results.filter(r => r.success).length, results });
});

app.post('/auth/join/:token', async (req, res) => {
  const supabaseAdmin = createClient(
    process.env.SUPABASE_URL,
    process.env.SUPABASE_SERVICE_ROLE_KEY
  );
  const { name, password } = req.body;
  try {
    // Verify token is still valid
    const { data: invite } = await supabaseAdmin
      .from('club_invites')
      .select('*, clubs(id, name)')
      .eq('token', req.params.token)
      .eq('status', 'pending')
      .single();
    if (!invite) return res.redirect(`/join/${req.params.token}?error=invalid`);
    if (new Date(invite.expires_at) < new Date()) {
      return res.redirect(`/join/${req.params.token}?error=expired`);
    }
    const email = invite.email === 'open-invite@realarenas.com'
      ? req.body.email
      : invite.email;
    if (!email) return res.redirect(`/join/${req.params.token}?error=missing_email`);
    // Create the user account
    const { data: authData, error: authError } = await supabaseAdmin
      .auth.admin.createUser({
        email,
        password,
        email_confirm: true,
        user_metadata: { name }
      });
    if (authError) {
      console.log('Join auth error:', authError.message);
      return res.redirect(`/join/${req.params.token}?error=account_exists`);
    }
    const userId = authData.user.id;
    // Create membership
    await supabaseAdmin.from('memberships').insert({
      user_id: userId,
      club_id: invite.club_id,
      role: invite.role || 'member',
      status: 'active'
    });
    // Mark invite as accepted
    await supabaseAdmin
      .from('club_invites')
      .update({ status: 'accepted', accepted_at: new Date().toISOString() })
      .eq('token', req.params.token);
    // Notify the club admin
    const { data: admins } = await supabaseAdmin
      .from('memberships')
      .select('user_id')
      .eq('club_id', invite.club_id)
      .eq('role', 'admin');
    for (const admin of (admins || [])) {
      await createNotification(supabaseAdmin, {
        userId: admin.user_id,
        type: 'club',
        title: 'New member joined',
        body: `${name} just accepted your invite and joined ${invite.clubs?.name}`,
        link: '/clubs/dashboard',
        actorId: userId,
        entityId: invite.club_id
      });
    }
    // Sign them in and redirect to feed
    const { data: signInData } = await supabaseAdmin
      .auth.signInWithPassword({ email, password });
    if (signInData?.session) {
      res.cookie('sb-access-token', signInData.session.access_token, {
        httpOnly: true,
        secure: process.env.NODE_ENV === 'production',
        sameSite: 'lax',
        maxAge: 7 * 24 * 60 * 60 * 1000
      });
    }
    res.redirect('/');
  } catch (err) {
    console.log('Join error:', err.message);
    res.redirect(`/join/${req.params.token}?error=unknown`);
  }
});

  // Send single invite
  window.sendSingleInvite = async function() {
    const emailEl = document.getElementById('invite-email');
    const roleEl = document.getElementById('invite-role');
    const email = emailEl?.value?.trim();
    const role = roleEl?.value || 'member';
    if (!email || !email.includes('@')) {
      showToast('Please enter a valid email address');
      return;
    }
    const btn = document.getElementById('send-invite-btn');
    if (btn) { btn.textContent = 'Sending…'; btn.disabled = true; }
    try {
      const response = await fetch(`/api/clubs/${clubId}/invite`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, role })
      });
      const result = await response.json();
      if (result.error) {
        showToast('Error: ' + result.error);
        if (btn) { btn.textContent = 'Send invite'; btn.disabled = false; }
        return;
      }
      showToast(`✓ Invite sent to ${email} — join link: ${result.invite?.joinUrl || 'check logs'}`);
      if (emailEl) emailEl.value = '';
      if (btn) { btn.textContent = 'Send invite'; btn.disabled = false; }
      loadInviteData();
    } catch (err) {
      showToast('Something went wrong — please try again');
      if (btn) { btn.textContent = 'Send invite'; btn.disabled = false; }
    }
  };
