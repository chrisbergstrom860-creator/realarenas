javascript(function buildManagedClubsMenu() {
  const data = window.ARENAS_DATA;
  const clubs = (data && data.clubs) || [];
  const managed = clubs.filter(c => c.role === 'admin' || c.role === 'coach');
  if (managed.length === 0) return; // pure athletes see nothing extra
  const menu = document.getElementById('userMenu');
  if (!menu) return;
  const sportIcons = { running:'🏃', cycling:'🚴', climbing:'🧗', swimming:'🏊', football:'⚽', weightlifting:'🏋️', hiking:'🥾', yoga:'🧘' };
  const sportBgs = { running:'#FFF7ED', cycling:'#EFF6FF', climbing:'#F5F3FF', swimming:'#F0FDFA', football:'#ECFDF5', weightlifting:'#FEF9C3', hiking:'#FAEEDA', yoga:'#FBEAF0' };
  const section = document.createElement('div');
  section.innerHTML =
    '<div style="font-size:10px;color:var(--gray-400);text-transform:uppercase;letter-spacing:.05em;padding:9px 14px 5px">Clubs you manage</div>' +
    managed.map(c => {
      const icon = sportIcons[c.sport] || '🏟';
      const bg = sportBgs[c.sport] || '#FFF7ED';
      return '<div class="menu-club-item" onclick="nav(\'/clubs/dashboard?club=' + c.id + '\')" style="display:flex;align-items:center;gap:10px;padding:9px 14px;cursor:pointer">' +
        '<div style="width:26px;height:26px;border-radius:7px;background:' + bg + ';display:flex;align-items:center;justify-content:center;font-size:13px;flex-shrink:0">' + icon + '</div>' +
        '<div style="flex:1;min-width:0"><div style="font-size:13px;font-weight:600;color:var(--gray-900);line-height:1.2;white-space:nowrap;overflow:hidden;text-overflow:ellipsis">' + c.name + '</div><div style="font-size:10px;color:#854D0E">Coach dashboard</div></div>' +
        '<span style="font-size:9px;font-weight:600;padding:2px 8px;border-radius:20px;background:#FAEEDA;color:#633806;flex-shrink:0">Manage</span>' +
      '</div>';
    }).join('') +
    '<div style="height:0.5px;background:var(--gray-200);margin:4px 0"></div>';
  // Insert at the very top of the menu, above existing items
  menu.insertBefore(section, menu.firstChild);
  // Add hover effect to match other menu items
  section.querySelectorAll('.menu-club-item').forEach(el => {
    el.addEventListener('mouseenter', () => el.style.background = 'var(--gray-50)');
    el.addEventListener('mouseleave', () => el.style.background = 'transparent');
  });
})();
