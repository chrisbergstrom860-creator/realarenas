Fix 3 — Add the JavaScript to wire the dropdown
Add this to the dashboard script:
javascriptlet notificationsPanelOpen = false;

window.toggleNotificationsPanel = async function() {
  const panel = document.getElementById('notifications-panel');
  if (!panel) return;
  notificationsPanelOpen = !notificationsPanelOpen;
  panel.style.display = notificationsPanelOpen ? 'flex' : 'none';
  if (notificationsPanelOpen) {
    loadNotificationsPanel();
  }
};

// Close panel when clicking outside
document.addEventListener('click', function(e) {
  const panel = document.getElementById('notifications-panel');
  const bell = document.querySelector('[onclick*="toggleNotificationsPanel"]');
  if (panel && !panel.contains(e.target) && bell && !bell.contains(e.target)) {
    panel.style.display = 'none';
    notificationsPanelOpen = false;
  }
});

async function loadNotificationsPanel() {
  const list = document.getElementById('notifications-panel-list');
  if (!list) return;
  try {
    const response = await fetch('/api/notifications');
    const data = await response.json();
    if (data.error || !data.notifications?.length) {
      list.innerHTML = '<div style="text-align:center;padding:32px;font-size:13px;color:var(--gray-400)">No notifications yet</div>';
      return;
    }
    const typeIcons = {
      like: '👍', follow: '👤', comment: '💬',
      club: '🏃', challenge: '⚡', event: '📅', system: '✦'
    };
    function timeAgo(dateStr) {
      const seconds = Math.floor((new Date() - new Date(dateStr)) / 1000);
      if (seconds < 60) return 'Just now';
      if (seconds < 3600) return Math.floor(seconds/60) + 'm ago';
      if (seconds < 86400) return Math.floor(seconds/3600) + 'h ago';
      return Math.floor(seconds/86400) + 'd ago';
    }
    list.innerHTML = data.notifications.slice(0, 15).map(n => `
      <div onclick="markNotificationRead('${n.id}')" style="display:flex;align-items:flex-start;gap:10px;padding:11px 16px;border-bottom:var(--border);cursor:pointer;transition:background .1s;background:${n.read ? 'white' : '#FFFDF0'}"
           onmouseenter="this.style.background='var(--gray-50)'"
           onmouseleave="this.style.background='${n.read ? 'white' : '#FFFDF0'}'">
        <div style="width:32px;height:32px;border-radius:50%;background:var(--yellow-light);display:flex;align-items:center;justify-content:center;font-size:14px;flex-shrink:0;margin-top:1px">
          ${typeIcons[n.type] || '✦'}
        </div>
        <div style="flex:1;min-width:0">
          <div style="font-size:12px;color:var(--gray-900);line-height:1.5;margin-bottom:2px">${n.body}</div>
          <div style="font-size:10px;color:var(--gray-400)">${timeAgo(n.created_at)}</div>
        </div>
        ${!n.read ? '<div style="width:7px;height:7px;border-radius:50%;background:#FFD21E;flex-shrink:0;margin-top:5px"></div>' : ''}
      </div>`
    ).join('');
    // Update unread badge
    const badge = document.querySelector('.notif-badge, .bell-badge, #notif-count');
    if (badge) badge.textContent = data.unreadCount > 0 ? data.unreadCount : '';
  } catch (err) {
    list.innerHTML = '<div style="text-align:center;padding:32px;font-size:13px;color:var(--gray-400)">Could not load notifications</div>';
  }
}

window.markNotificationRead = async function(id) {
  await fetch('/api/notifications/' + id + '/read', { method: 'POST' });
  const item = document.querySelector(`[onclick*="${id}"]`);
  if (item) {
    item.style.background = 'white';
    const dot = item.querySelector('[style*="background:#FFD21E"]');
    if (dot) dot.remove();
  }
};

window.markAllNotificationsRead = async function() {
  await fetch('/api/notifications/read-all', { method: 'POST' });
  document.querySelectorAll('#notifications-panel-list [style*="FFFDF0"]').forEach(item => {
    item.style.background = 'white';
    const dot = item.querySelector('[style*="background:#FFD21E"]');
    if (dot) dot.remove();
  });
  const badge = document.querySelector('.notif-badge, .bell-badge, #notif-count');
  if (badge) badge.textContent = '';
  showToast('All notifications marked as read');
};

// Load notification count on page load
async function loadDashboardNotifCount() {
  try {
    const response = await fetch('/api/notifications');
    const data = await response.json();
    if (data.unreadCount > 0) {
      const badge = document.querySelector('.notif-badge, .bell-badge, #notif-count');
      if (badge) badge.textContent = data.unreadCount;
    }
  } catch (err) {}
}
